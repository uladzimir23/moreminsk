## TypeScript-тип

Будет лежать в `src/entities/yacht/model/types.ts`:

```typescript
export type YachtType = 'sail' | 'motor' | 'sail-motor';

export interface Yacht {
  slug: string;
  name: string;
  type: YachtType;
  capacity: number;
  lengthMeters?: number;
  pricePerHour: number;
  minHours: number;
  description: string;
  features: string[];
  suitableFor: string[];  // service slugs
  gallery: string[];
  mainImage: string;
  video?: string;
}
```
